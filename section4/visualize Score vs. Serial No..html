import json
import matplotlib.pyplot as plt
import random

with open(r"C:\Users\Xiuhao Ding\Downloads\finalresults.txt", 'r', encoding='utf-8') as file2:    
    res_ = file2.readlines()

i = 0
x = []
y = []
z = []

while i < len(res_):
    res_[i] = res_[i].split('\t')
    res_[i][2] = res_[i][2].strip('\n')
    x.append(i)
    y.append(float(res_[i][1]))
    z.append(float(res_[i][2]))
    i+=1
# plt.figure(figsize=(20, 8), dpi=100)
plt.xlabel('Serial No.')
plt.ylabel('c')
plt.scatter(x, y, color='r')
plt.scatter(x, z, color='b')
plt.show()

print('done')
