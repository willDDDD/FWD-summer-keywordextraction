import json
import matplotlib.pyplot as plt
import random

with open(r"C:\Users\Xiuhao Ding\Downloads\sortavg.txt", 'r', encoding='utf-8') as file2:    
    res_ = file2.readlines()

i = 0
x = []
y = []

while i < len(res_):
    res_[i] = res_[i].split('\t')
    res_[i][3] = res_[i][3].strip('\n')
    x.append(float(res_[i][2]))
    y.append(float(res_[i][3]))
    i+=1
# plt.figure(figsize=(20, 8), dpi=100)
plt.xlabel('important score')
plt.ylabel('relevance score')
plt.scatter(x, y)
plt.show()

print('done')
